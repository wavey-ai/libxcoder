435 bool find_available_guid(ni_device_queue_t *device_queue, int device_type, int *guidn)
436 {
437  /*
438  Create a mask of all 128 guids, mark used by 1 and unused by 0.
439  */
440  int32_t temp_guid;
441  int32_t i, j;
442  uint32_t guid_mask[4] = {0};
443 
444  for (i = 0; i < NI_MAX_DEVICE_CNT; i++)
445  {
446  temp_guid = device_queue->xcoders[device_type][i];
447  if (temp_guid >= 0 && temp_guid < NI_MAX_DEVICE_CNT)
448  {
449  guid_mask[temp_guid / 32] |= (1u << ((uint32_t)temp_guid % 32));
450  }
451  }
452  //from the masks find the first available guidn
453  for (i = 0; i < 4; i++)
454  {
455  for (j = 0; j < 32; j++)
456  {
457  if ((guid_mask[i] & (1u << j)) == 0)
458  {
459  *guidn = (i * 32) + j;
460  return true;
461  }
462  }
463  }
464 
465  return false;
466 }

468 bool add_to_shared_memory(const char device_name[NI_MAX_DEVICE_NAME_LEN],
469  const bool device_open_should_succeed,
470  const int should_match_rev,
471  ni_device_queue_t *device_queue)
472 {

477 
478  ni_device_capability_t device_capability;
479  ni_device_handle_t device_handle;
480  ni_device_info_t device_info = {0};
481  bool success = true;
482

484  device_handle = ni_device_open(device_name, &max_io_size);
485  if (device_handle == NI_INVALID_DEVICE_HANDLE)
486  {

488  {
489  ni_log(NI_LOG_ERROR,

491  __FUNCTION__,
492  device_name);
493  return false;
494  }
495  return true;
496  }
497 
498  if (!check_device_capability(device_handle,
499  &device_capability,
500  device_name,
501  should_match_rev,
502  &compatibility))
503  {
504  LRETURN;
505  }
506 
507  if (compatibility == 2)
508  {
509  ni_log(NI_LOG_INFO,
510  "WARNING: %s with FW revision %.*s is below the minimum "

513  device_name,
514  (int)sizeof(device_capability.fw_rev),
515  device_capability.fw_rev);
516  }
517 
518 
519  for (i = NI_DEVICE_TYPE_DECODER; i < NI_DEVICE_TYPE_XCODER_MAX; i++)
520  {
521  if (!device_capability.xcoder_cnt[i])
522  {
523  ni_log(NI_LOG_DEBUG,
524  "%s %s disabled...\n",
525  device_name,
526  GET_XCODER_DEVICE_TYPE_STR(i));
527  continue;
528  }
529 
530  j = device_queue->xcoder_cnt[i];
531  guid = j ? device_queue->xcoders[i][j-1] + 1 : 0;
532  if (guid >= NI_MAX_DEVICE_CNT && !find_available_guid(device_queue, i, &guid))
533  {
534  ni_log(NI_LOG_ERROR,
535  "ERROR: %s(): Failed to add %s: too many devices\n",
536  __FUNCTION__,
537  device_name);
538  success = false;
539  LRETURN;
540  }
541  device_queue->xcoders[i][j] = guid;
542  device_queue->xcoder_cnt[i]++;
543 
544  fill_device_info(&device_info,
545  guid,
546  device_handle,
547  device_name,
548  compatibility,
549  &device_capability,
550  (ni_device_type_t)i);
551 
552  ni_rsrc_get_one_device_info(&device_info);
553  }
554 
555 END:
556  ni_device_close(device_handle);
557 
558  return success;
