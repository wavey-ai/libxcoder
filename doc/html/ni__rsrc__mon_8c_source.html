519 /*!*****************************************************************************
520  * \brief Accumulate string data in a dynamically sized buffer. This is
521  * useful to separate error messages from json and table output.
522  *
523  * \param[in] *dyn_str_buf pointer to structure holding dyn_str_buf info
524  * \param[in] *fmt printf format specifier
525  * \param[in] ... additional arguments
526  *
527  * \return 0 for success, -1 for error
528  ******************************************************************************/
529 int strcat_dyn_buf(dyn_str_buf_t *dyn_str_buf, const char *fmt, ...)
530 {
531  int avail_buf;
532  int formatted_len;
533  int add_buf_size = 0;
534  char *tmp_char_ptr = NULL;
535 
536  if (!dyn_str_buf)
537  {
538  fprintf(stderr, "ERROR: invalid param *dyn_str_buf\n");
539  return -1;
540  }
541 
542  if (!fmt)
543  {
544  return 0;
545  }
546 
547  va_list vl, tmp_vl;
548  va_start(vl, fmt);
549  va_copy(tmp_vl, vl);
550 
551  // determine length of string to add
552  formatted_len = vsnprintf(NULL, 0, fmt, tmp_vl);
553  va_end(tmp_vl);
554 
555  // check if str_buf needs to be expanded in increments of chunk size
556  avail_buf = dyn_str_buf->buf_size - dyn_str_buf->str_len;
557  add_buf_size = (formatted_len + 1) > avail_buf ?
558  ((formatted_len + 1 - avail_buf + DYN_STR_BUF_CHUNK_SIZE - 1) /
559  DYN_STR_BUF_CHUNK_SIZE) *
560  DYN_STR_BUF_CHUNK_SIZE :
561  0;
562 
563  // realloc() to expand str_buf if necessary
564  if (add_buf_size)
565  {
566  tmp_char_ptr = (char *)realloc(dyn_str_buf->str_buf,
567  sizeof(char) * dyn_str_buf->buf_size +
568  add_buf_size);
569  if (!tmp_char_ptr)
570  {
571  fprintf(stderr, "ERROR: strcat_dyn_buf() failed realloc()\n");
572  va_end(vl);
573  return -1;
574  }
575  dyn_str_buf->str_buf = tmp_char_ptr;
576  dyn_str_buf->buf_size += add_buf_size;
577  }
578 
579  // concatenate string to buffer
580  vsprintf(dyn_str_buf->str_buf + dyn_str_buf->str_len, fmt, vl);
581  dyn_str_buf->str_len += formatted_len;
582 
583  va_end(vl);
584  return 0;
585 }
