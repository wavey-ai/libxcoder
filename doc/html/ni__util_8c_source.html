132  *
133  * \param[in/out] p_tp timeval struct

135  *
136  * \return return 0 for success, -1 for error

138 int32_t ni_gettimeofday(struct timeval *p_tp, void *p_tzp)
139 {
140 #ifdef _WIN32
141  FILETIME file_time;
142  SYSTEMTIME system_time;
143  ULARGE_INTEGER ularge;
145  static const unsigned __int64 epoch =
146  ((unsigned __int64)116444736000000000ULL);
147 
148  // timezone information is stored outside the kernel so tzp isn't used
149  (void *)p_tzp;
151  // Note: this function is not a precision timer. See elapsed_time().
152  GetSystemTime(&system_time);
153  SystemTimeToFileTime(&system_time, &file_time);
154  ularge.LowPart = file_time.dwLowDateTime;
155  ularge.HighPart = file_time.dwHighDateTime;
156  // Surpress cppcheck
157  (void)ularge.LowPart;
158  (void)ularge.HighPart;
159  p_tp->tv_sec = (long)((ularge.QuadPart - epoch) / 10000000L);
160  p_tp->tv_usec = (long)(system_time.wMilliseconds * 1000);
161 
162  return 0;
163 #else
164  return gettimeofday(p_tp, p_tzp);

168 uint32_t ni_round_up(uint32_t number_to_round, uint32_t multiple)
169 {
170  if (0 == multiple)
171  {
172  return number_to_round;
173  }
174 
175  uint32_t remainder = number_to_round % multiple;
176  if (0 == remainder)
177  {
178  return number_to_round;
179  }
180 
181  return (number_to_round + multiple - remainder);

348  QueryPerformanceCounter(&StartCount);

350  StopCount.QuadPart =
351  StartCount.QuadPart + usec * (Frequency.QuadPart / 1000000);
352  do
353  {
354  QueryPerformanceCounter(&StartCount);
355  } while (StartCount.QuadPart < StopCount.QuadPart);

365 
366  timer = CreateWaitableTimer(NULL, TRUE, NULL);
367  if (NULL != timer)

403 {

449 // get a free memory buffer from the pool
450 ni_buf_t *ni_buf_pool_get_buffer(ni_buf_pool_t *p_buffer_pool)
451 {
452  ni_buf_t *buf = NULL;
453 
454  if (NULL == p_buffer_pool)
455  {
456  return NULL;
457  }
458 
459  ni_pthread_mutex_lock(&p_buffer_pool->mutex);
460  buf = p_buffer_pool->p_free_head;
461 
462  // find and return a free buffer
463  if (NULL == buf)
464  {
465  ni_log(NI_LOG_INFO, "Expanding dec fme buffer_pool from %u to %u\n",
466  p_buffer_pool->number_of_buffers,
467  p_buffer_pool->number_of_buffers +
468  NI_DEC_FRAME_BUF_POOL_SIZE_EXPAND);
469 
470  buf = ni_buf_pool_expand(p_buffer_pool);
471  if (NULL == buf)
472  {
473  ni_pthread_mutex_unlock(&p_buffer_pool->mutex);
474  return NULL;
475  }
476  }
477 
478  // remove it from free list head; reconnect the linked list, the p_next
479  // will become the new head now
480  p_buffer_pool->p_free_head = buf->p_next_buffer;
481 
482  if (NULL != buf->p_next_buffer)
483  {
484  buf->p_next_buffer->p_previous_buffer = NULL;
485  } else
486  {
487  p_buffer_pool->p_free_tail = NULL;
488  }
489 
490  // add it to the used list tail
491  buf->p_previous_buffer = p_buffer_pool->p_used_tail;
492  buf->p_next_buffer = NULL;
493 
494  if (NULL != p_buffer_pool->p_used_tail)
495  {
496  p_buffer_pool->p_used_tail->p_next_buffer = buf;
497  } else
498  {
499  p_buffer_pool->p_used_head = buf;
500  }
501 
502  p_buffer_pool->p_used_tail = buf;
503 
504  ni_pthread_mutex_unlock(&p_buffer_pool->mutex);
505 
506  ni_log(NI_LOG_DEBUG, "%s ptr %p buf %p\n", __func__, buf->buf, buf);
507  return buf;
508 }

1070 {

1073  uint32_t matched_chr; // boolean flag for when match is found for a character in the pattern
1074  char char_match_pattern[11] = ""; // what characters to look for when evaluating a character in main_str

1080  if (!main_str || !pattern || !*main_str || !*pattern)
1081  {
1082  return 0;
1083  }

1086  for (i = 0; i < strlen(main_str) && !pattern_matched; i++)
1087  {
1088  pattern_matched = 0;
1089  match_length = 0;
1090  // iterate over the characters of the pattern
1091  for (j = 0; j < strlen(pattern); j++)
1092  {
1093  matched_chr = 0;

1095  if (pattern[j] == '+')

1098  return 0;

1101  {
1102  memcpy(char_match_pattern, digit_match_pattern, strlen(digit_match_pattern) + 1);
1103  }
1104  else
1105  {
1106  memcpy(char_match_pattern, pattern + j, 1);
1107  memset(char_match_pattern + 1, 0, 1);
1108  }
1109  // check if char is in match_pattern
1110  if (pattern[j+1] == '+')
1111  {
1112  while (main_str[i + match_length] && strchr(char_match_pattern, (int) main_str[i + match_length]))
1113  {
1114  match_length++;
1115  matched_chr = 1;
1116  }
1117  j++;
1118  }
1119  else if (main_str[i + match_length] && strchr(char_match_pattern, (int) main_str[i + match_length]))
1120  {
1121  match_length++;
1122  matched_chr = 1;
1123  }
1124  // if no matches were found, then this segment is not the sought pattern
1125  if (!matched_chr)
1126  {
1127  break;
1128  }
1129  // if all of pattern has been processed and matched, mark sucessful whole pattern match
1130  else if ((j + 1) >= strlen(pattern))
1131  {
1132  pattern_matched = 1;
1133  pattern_start = i;
1134  }
1135  }
1136  }
1137 
1138  // remove sub-string if its pattern was found in main_str
1139  if (pattern_matched)
1140  {
1141  uint32_t orig_main_str_len = (uint32_t)strlen(main_str);
1142  memmove(main_str + pattern_start, main_str + pattern_start + match_length,
1143  strlen(main_str + pattern_start + match_length));

1146  }

1148  {
1149  return 0;
1150  }
